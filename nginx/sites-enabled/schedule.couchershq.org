server {
    listen 80;
    server_name schedule.couchershq.org;
    return 301 https://schedule.couchershq.org$request_uri;
}

server {
    listen 443 ssl http2;
    server_name schedule.couchershq.org;

    access_log /var/log/nginx/calendso.access.log;
    error_log /var/log/nginx/calendso.error.log;

    ssl_certificate /certs/live/couchershq.org/fullchain.pem;
    ssl_certificate_key /certs/live/couchershq.org/privkey.pem;

    location / {
        client_max_body_size 0;

        proxy_set_header    Host                $http_host;
        proxy_set_header    X-Real-IP           $remote_addr;
        proxy_set_header    X-Forwarded-Ssl     on;
        proxy_set_header    X-Forwarded-For     $proxy_add_x_forwarded_for;
        proxy_set_header    X-Forwarded-Proto   $scheme;

        proxy_pass http://calendso:3000;
    }
}
